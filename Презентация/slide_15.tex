    \begin{tcolorbox}
        \linespread{1}
        \begin{verbatim}
            
\lstset{
    % Язык программирования
        language=Python,
    % С какой стороны нумеровать
        numbers=left,
    % Стиль который будет использоваться
    % для нумерации строк
        numberstyle=\tiny\color{gray},
    % Шаг между линиями. Если 1, то будет
    % пронумерована каждая строка
        stepnumber=2,
        numbersep=5pt,
    ...


\end{verbatim}
    \end{tcolorbox}
    
\end{frame}

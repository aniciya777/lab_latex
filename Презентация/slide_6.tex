    \begin{tcolorbox}
        \linespread{1}
        \begin{verbatim}
            
            \begin{lstlisting}
                Вставьте Ваш код здесь
            \end{lstlisting}
        \end{verbatim}
    \end{tcolorbox}
    
\end{frame}
